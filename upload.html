// ================================= // FashionAI Hybrid Clothing Analyzer // clothing-ai.js // ================================= import { askGemini } from "./gemini-ai.js"; import { localAnalyze } from "./local-fashion-ai.js"; export async function analyzeClothing(image){ console.log( "🤖 Hybrid FashionAI Started" ); // ================================= // 1. Local AI FIRST // Always available // ================================= let localResult = localAnalyze(image); console.log( "LOCAL RESULT:", localResult ); // ================================= // 2. Try Gemini Enhancement // ================================= try{ const prompt = ` You are FashionAI clothing analyzer. Analyze this clothing image. Return ONLY valid JSON. Format: { "type":"", "category":"", "primaryColor":"", "secondaryColor":"", "material":"", "style":"", "occasion":"", "matchingItems":[] } `; const response = await askGemini( prompt, image ); const cloudResult = JSON.parse(response); console.log( "GEMINI RESULT:", cloudResult ); // Combine Local + Gemini return { ...localResult, ...cloudResult, analyzedBy: "Hybrid AI + Gemini" }; } catch(error){ console.log( "Gemini unavailable. Using Local FashionAI" ); // ================================= // Offline mode // ================================= return { ...localResult, category: localResult.category || "General Wear", occasion: localResult.occasion || "Flexible", matchingItems: [], analyzedBy: "Offline Hybrid FashionAI" }; } }
